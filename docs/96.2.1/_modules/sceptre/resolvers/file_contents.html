[docs]    def resolve(self):
        """
        Retrieves the contents of a file at a given absolute file path.

        :returns: Contents of file.
        :rtype: str
        """
        try:
            with open(self.argument, "r") as file:
                return file.read()
        except (EnvironmentError, TypeError) as e:
            raise e
